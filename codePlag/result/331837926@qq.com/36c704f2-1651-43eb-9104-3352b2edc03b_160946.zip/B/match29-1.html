int a[30];
int minn(int i,int j)
{
    int ans = a[i];
    for (;i<=j;i++) ans = min(ans,a[i]);
    return ans;
}
int main()
{

    int T;

    while (T--)
    {
        int ans = 0;
        int temp;
        scanf("%d",&temp);
        for (int i  = 1;i<=12;i++)
            scanf("%d",&a[i]);
        for (int i = 2;i<=11;i++)
            for (int j = i;j<=11;j++)
        {

            if (minn(i,j)>a[i-1]&&minn(i,j)>a[j+1])
            {
                ans++;
            }
        }
        cout<<temp<<' '<<ans<<endl;
    }
    return 0;
}
