int p,q,cnt;
void cal()
{

    cnt=0;

    while(p>q)
    {
        p=p-q;
        cnt++;

    }

    int i,j;
    i=p;
    j=q;
    p=j;
    q=j-i;

    while(cnt>0)
    {
        cnt--;
        q=p+q;
    }
}
