int a[15];

int ncase,k,res;

bool check(int s,int e){
	int goal=max(a[s-1],a[e+1]);
	for(int i=s;i<=e;i++){
		if(goal>=a[i]) return false;
	}
	return true;
}

int main()
{
	scanf("%d",&ncase);
	while(ncase--){
		scanf("%d",&k);
		for(int i=0;i<12;i++){
			scanf("%d",&a[i]);
		}
		res=0;
		for(int i=1;i<11;i++){
			for(int j=i;j<11;j++){
				if(check(i,j)){
					res++;
				}
			}
		}
		printf("%d %d\n",k,res);
	}
	return 0;
}
