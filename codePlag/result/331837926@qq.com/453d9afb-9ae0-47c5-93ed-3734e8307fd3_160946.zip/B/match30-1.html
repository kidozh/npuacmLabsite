int a[25];

int main()
{
    int t;
    scanf("%d",&t);
    while (t --)
    {
        int k;
        scanf("%d",&k);
        for (int i = 0 ;i < 12; ++ i)
            scanf("%d",&a[i]);

        int sum = 0;
        for (int i = 1;i < 11; ++ i)
        {
            for (int j = i;j < 11; ++ j)
            {
                bool flag = 1;
                for (int k = i;k <= j; ++ k)
                {
                    if (a[k] <= a[i - 1] || a[k] <= a[j + 1])
                    {
                        flag = 0;
                        break;
                    }
                }
                sum += flag;
            }
        }
        printf("%d %d\n",k,sum);
    }
    return 0;
}
